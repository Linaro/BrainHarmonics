99  {

101  {
102  if(val.size() > 2)

105  {
106  if(val[reduction_loop] < -1)
107  {
108  val[reduction_loop] = (2 / val[reduction_loop]) + 1;
109  }
110  if(val[reduction_loop] > 1)
111  {
112  val[reduction_loop] = (2 / val[reduction_loop]) - 1;
113  }
114  }
115  }
116  }
