48 Sniffex::print_hex_ascii_line(const u_char *payload, int len, int offset)
49 {
50 
51  int i;
52  int gap;
53  const u_char *ch;
54

56  printf("%05d ", offset);

58  /* hex */

69  printf(" ");
70

72  if (len < 16) {
73  gap = 16 - len;
74  for (i = 0; i < gap; i++) {
75  printf(" ");
76  }
77  }

95 /*
96  * print packet payload data (avoid printing binary data)
97  */
98 void
99 Sniffex::print_payload(const u_char *payload, int len)
100 {
101 
102  int len_rem = len;
103  int line_width = 16; /* number of bytes per line */
104  int line_len;
105  int offset = 0; /* zero-based offset counter */
106  const u_char *ch = payload;
107 
108  if (len <= 0)
109  return;
110 
111  /* data fits on one line */
112  if (len <= line_width) {
113  print_hex_ascii_line(ch, len, offset);
114  return;
115  }
116 
117  /* data spans multiple lines */
118  for ( ;; ) {
119  /* compute current line length */
120  line_len = line_width % len_rem;
121  /* print line */
122  print_hex_ascii_line(ch, line_len, offset);
123  /* compute total remaining */
124  len_rem = len_rem - line_len;
125  /* shift pointer to remaining bytes to print */
126  ch = ch + line_len;
127  /* add offset */
128  offset = offset + line_width;
129  /* check if we have line width chars or less */
130  if (len_rem <= line_width) {
131  /* print last line and Get out */
132  print_hex_ascii_line(ch, len_rem, offset);
133  break;
134  }
135  }
136 
137  return;
138 }

144 Sniffex::got_packet(u_char *args, const struct pcap_pkthdr *header, const u_char *packet)
145 {
146 
147  static int count = 1; /* packet counter */
148 
149  /* declare pointers to packet headers */
150  const struct sniff_ethernet *ethernet; /* The ethernet header [1] */
151  const struct sniff_ip *ip; /* The IP header */
152  const struct sniff_tcp *tcp; /* The TCP header */
153  const u_char *payload; /* Packet payload */
154 
155  int size_ip;
156  int size_tcp;
157  int size_payload;
158 
159  printf("\nPacket number %d:\n", count);
160  count++;
161 
162  /* define ethernet header */
163  ethernet = (struct sniff_ethernet*)(packet);
164 
165  /* define/compute ip header offset */
166  ip = (struct sniff_ip*)(packet + SIZE_ETHERNET);
167  size_ip = IP_HL(ip)*4;
168  if (size_ip < 20) {
169  printf(" * Invalid IP header length: %u bytes\n", size_ip);
170  return;
171  }
172 
173  /* print source and destination IP addresses */
174  printf(" From: %s\n", inet_ntoa(ip->ip_src));
175  printf(" To: %s\n", inet_ntoa(ip->ip_dst));
176 
177  /* determine protocol */
178  switch(ip->ip_p) {
179  case IPPROTO_TCP:
180  printf(" Protocol: TCP\n");
181  break;
182  case IPPROTO_UDP:
183  printf(" Protocol: UDP\n");
184  return;
185  case IPPROTO_ICMP:
186  printf(" Protocol: ICMP\n");
187  return;
188  case IPPROTO_IP:
189  printf(" Protocol: IP\n");
190  return;
191  default:
192  printf(" Protocol: unknown\n");
193  return;
194  }
195 
196  /*
197  * OK, this packet is TCP.
198  */
199 
200  /* define/compute tcp header offset */
201  tcp = (struct sniff_tcp*)(packet + SIZE_ETHERNET + size_ip);
202  size_tcp = TH_OFF(tcp)*4;
203  if (size_tcp < 20) {
204  printf(" * Invalid TCP header length: %u bytes\n", size_tcp);
205  return;
206  }
207 
208  printf(" Src port: %d\n", ntohs(tcp->th_sport));
209  printf(" Dst port: %d\n", ntohs(tcp->th_dport));
210 
211  /* define/compute tcp payload (segment) offset */
212  //payload = (char *)(packet + SIZE_ETHERNET + size_ip + size_tcp);
213  payload = (u_char *)(packet + SIZE_ETHERNET + size_ip + size_tcp);
214 
215  /* compute tcp payload (segment) size */
216  size_payload = ntohs(ip->ip_len) - (size_ip + size_tcp);
217 
218  /*
219  * Print payload data; it might be binary, so don't just
220  * treat it as a string.
221  */
222  if (size_payload > 0) {
223  printf(" Payload (%d bytes):\n", size_payload);
224  print_payload(payload, size_payload);
225  }
226 
227  return;
228 }
