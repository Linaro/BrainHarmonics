104 struct SquareGrid {
105  typedef tuple<int,int> Location;
106  static array<Location, 4> DIRS;
107 
108  int width, height;
109  unordered_set<Location> walls;
110 
111  SquareGrid(int width_, int height_)
112  : width(width_), height(height_) {}
113 
114  inline bool in_bounds(Location id) {
115  int x, y;
116  tie (x, y) = id;
117  return 0 <= x && x < width && 0 <= y && y < height;
118  }
119 
120  inline bool passable(Location id) {
121  return !walls.count(id);
122  }
123 
124  vector<Location> neighbors(Location id) {
125  int x, y, dx, dy;
126  tie (x, y) = id;
127  vector<Location> results;
128 
129  for (auto dir : DIRS) {
130  tie (dx, dy) = dir;
131  Location next(x + dx, y + dy);
132  if (in_bounds(next) && passable(next)) {
133  results.push_back(next);
134  }
135  }
136 
137  if ((x + y) % 2 == 0) {
138  // aesthetic improvement on square grids
139  std::reverse(results.begin(), results.end());
140  }
141 
142  return results;
143  }
144 };

208 template<typename Graph>
209 void dijkstra_search
210  (Graph graph,
211  typename Graph::Location start,
212  typename Graph::Location goal,
213  unordered_map<typename Graph::Location, typename Graph::Location>& came_from,
214  unordered_map<typename Graph::Location, int>& cost_so_far)
215 {
216  typedef typename Graph::Location Location;
217  PriorityQueue<Location> frontier;
218  frontier.put(start, 0);
219 
220  came_from[start] = start;
221  cost_so_far[start] = 0;
222 
223  while (!frontier.empty()) {
224  auto current = frontier.get();
225 
226  if (current == goal) {
227  break;
228  }
229 
230  for (auto next : graph.neighbors(current)) {
231  int new_cost = cost_so_far[current] + graph.cost(current, next);
232  if (!cost_so_far.count(next) || new_cost < cost_so_far[next]) {
233  cost_so_far[next] = new_cost;
234  came_from[next] = current;
235  frontier.put(next, new_cost);
236  }
237  }
238  }
239 }

258 inline int heuristic(SquareGrid::Location a, SquareGrid::Location b) {
259  int x1, y1, x2, y2;
