53 Install the QT/VTK toolkit
54 
55 ```
56 cd ~/Developer

58 
59 mkdir ~/Developer/VTK-build
60 mkdir ~/Developer/VTK-Release-build
61 mkdir ~/Developer/QT

63 cmake ~/Developer/VTK
64 
65 ```
66 (Ubuntu element)
67 ```
68 cd ~/Developer/VTK-Release-build
69 cmake -DCMAKE_BUILD_TYPE:STRING=Release ~/Developer/VTK
70 
71 cd ~/Developer/QT
72 wget http://download.qt-project.org/official_releases/qt/5.9/5.9.2/qt-opensource-linux-x64-5.9.2.run
73 chmod +x qt-opensource-linux-x64-5.9.2.run
74 ./qt-opensource-linux-x64-5.9.2.run
75 
76 cd ~/Developer/VTK-Release-build
77 cmake -DVTK_QT_VERSION:STRING=5 \
78  -DQT_QMAKE_EXECUTABLE:PATH=~/Qt5.9.2/5.9.2/gcc_64/bin/qmake \
79  -DVTK_Group_Qt:BOOL=ON \
80  -DCMAKE_PREFIX_PATH:PATH=~/Qt5.9.2/5.9.2/gcc_64/lib/cmake \
81  -DBUILD_SHARED_LIBS:BOOL=ON \
82  ~/Developer/VTK
83 
84 make -j2
85 sudo make install
86 ```
