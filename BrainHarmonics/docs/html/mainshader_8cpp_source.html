93 {

105  if (fragment_shader != 0u)

108  program = glCreateProgram();
109  if (program != 0u)

111  /* attach both shader and link */
112  glAttachShader(program, vertex_shader);
113  glAttachShader(program, fragment_shader);
114  glLinkProgram(program);
115  glGetProgramiv(program, GL_LINK_STATUS, &program_ok);
116 
117  if (program_ok != GL_TRUE)
118  {
119  fprintf(stderr, "ERROR, failed to link shader program\n");
120  glGetProgramInfoLog(program, 8192, &log_length, info_log);
121  fprintf(stderr, "ERROR: \n%s\n\n", info_log);
122  glDeleteProgram(program);
123  glDeleteShader(fragment_shader);
124  glDeleteShader(vertex_shader);
125  program = 0u;
126  }
127  }

132  glDeleteShader(vertex_shader);

138  }
